Binding event on "section" element and delegates to "a" tag below.

let section = document.querySelector('section');

section.addDelegateListener('click', 'a', function handlerFn(event) {
    event.preventDefault();
    alert('link clicked!');
});

Click me!

If you want to remove the event:

let section = document.querySelector('section');

section.removeDelegateListener('click', 'a', handlerFn);
